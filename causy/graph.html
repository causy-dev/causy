431    def add_edge(self, u: Node, v: Node, metadata: Dict):
432        """
433        Add an edge to the graph
434        :param u: u node
435        :param v: v node
436        :param metadata: metadata of the edge
437        :return:
438        """
439
440        if u.id not in self.nodes:
441            raise GraphError(f"Node {u} does not exist")
442        if v.id not in self.nodes:
443            raise GraphError(f"Node {v} does not exist")
444
445        if u.id == v.id:
446            raise GraphError("Self loops are currently not allowed")
447
448        if u.id not in self.edges:

451        if v.id not in self.edges:
452            self.edges[v.id] = self.__init_dict()
453            self._reverse_edges[v.id] = self.__init_dict()
454
455        a_edge = Edge(u=u, v=v, edge_type=UndirectedEdge(), metadata=metadata)
456        self.edges[u.id][v.id] = a_edge
457        self._reverse_edges[v.id][u.id] = a_edge
458
459        b_edge = Edge(u=u, v=v, edge_type=UndirectedEdge(), metadata=metadata)
460        self.edges[v.id][u.id] = b_edge
461        self._reverse_edges[u.id][v.id] = b_edge
462
463        self.edge_history[(u.id, v.id)] = []
464        self.edge_history[(v.id, u.id)] = []
