 92    def add_edge(self, u: Node, v: Node, metadata: Dict):
 93        """
 94        Add an edge to the graph

 96        :param v: v node
 97        :param metadata: metadata of the edge
 98        :return:
 99        """
100
101        if u.id not in self.nodes:

105
106        if u.id == v.id:

445    def directed_path_exists(self, u: Node, v: Node):
446        """

459    def directed_paths(self, u: Node, v: Node):
460        """
461        Return all directed paths from u to v
462        :param u: node u
463        :param v: node v
464        :return: list of directed paths
465        """
466        # TODO: try a better data structure for this
467        if self.directed_edge_exists(u, v):
468            return [[(u, v)]]
469        paths = []
470        for w in self.edges[u.id]:
471            if self.directed_edge_exists(u, self.nodes[w]):
472                for path in self.directed_paths(self.nodes[w], v):
473                    paths.append([(u, self.nodes[w])] + path)

497                if not self.bidirected_edge_exists(r, w):
498                    # TODO: check if this is correct (@sof)
499                    return True
500        return False
